ColdFusion Code Sample

Authenticates User Login

            


        <!--- 
        This function checks the db to authenticate the user logging in. 
        Checks if username entered exixts from form, then checks against
        password.
        If authentication passes session variables are set. 
       If not, user is sent to error page 
      --->


                      
     <cfcomponent displayname="Authenticate" hint="Authenticates user login">
                      
        <cffunction name="Login" hint="checks login entered, compares to db">
                          
           <cfquery name= "Authenticate"  datasource = "#request.datasource#">
                SELECT
                    hashPassword,
                    saltPassword,
                    userRole,
                    user_ID
               FROM
                    Users
              WHERE CFSQL
                    userName = < cfqueryparam CFSQLType="CF_SQL_VARCHAR" 
                    value= "#form.username#" >
          </cfquery>
                        
                      
        <!--- Username doesn't exist, send to error page --->
           <cfif !Authenticate.recordcount>
          <cflocation  url="error.cfm"  addtoken=false>
          <cfif>
                      
                      
        <!--- If username exists, now test to see if password passes ----->
          <cfif Authenticate.hashPassword EQ hash(form.password & Authenticate.saltPassword, "SHA-512")>
                        
            <!--- user is authenticated, set the session variables ---->

               <cfset Session.Auth = StructNew()>
               <cfset Session.Auth.IsLoggedIn = "Yes">
               <cfset Session.Auth.UserID = Authenticate.user_ID>
              <cfset Session.Auth.UserRole = Authenticate.userRole>
              <cfset Session.Auth.userName = "#form.username#">
         
                      
          <cfelse> <!--- password failed, send to error page --->
              <cflocation  url="error.cfm" addtoken=false>
          </cfif>
                      
       <cfreturn Login >
       </cffunction>
      </cfcomponent>
